colpick Color Picker

Usage examples

Button

$('#example-button').colpick();

Flat mode and hex layout

$('#example-flat').colpick({ flat:true, layout:'hex' });

Text Input and rgbhex layout

$('#example-text').colpick({ layout:'rgbhex', onChange:function(hsb,hex,rgb,el,bySetColor) { $(el).val('#'+hex); } });

HTML5 Color Input

$('#example-color').colpick({ color:'#FF0000', onSubmit:function(hsb,hex,rgb,el,bySetColor) { $(el).val('#'+hex); $(el).colpickHide(); } });

Polyfill

$('#example-polyfill').colpick({ polyfill:true });
